CREATE TRIGGER check_book_issue BEFORE UPDATE ON BOOKISSUED 		
	FOR EACH ROW 		
	BEGIN 		
		IF DATE_ADD(NEW.ISSUEDATE, INTERVAL 21 DAY) < CURRENT_DATE() THEN 	
			SET NEW.NOTICEDATE = DATE();
		ELSEIF DATE_ADD(NEW.ISSUEDATE, INTERVAL 21 DAY) > CURRENT_DATE() THEN 	
			SET NEW.NOTICEDATE = NULL; 
		END IF; 	
	END;
